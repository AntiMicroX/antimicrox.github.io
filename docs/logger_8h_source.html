161  {
162  if (file == stdout)

164  m_message << "Printed stdout message📓: ";
165  } else if (file == stderr)
166  {
167  m_message << "Printed stderr message📓: ";
168  } else
169  {
170  m_message << "unknown stream ";
171  }
172  };
173 
174  ~StreamPrinter()
175  {
176  // When logger prints to stream, then we already have printed messages into console,
177  // there is no need to duplicate
178  if (Logger::getInstance()->isWritingToFile())
179  LogHelper(Logger::LogLevel::LOG_INFO, m_lineno, m_filename, QString(m_message.str().c_str())).sendMessage();
180  };
